Podstawowe komendy wyswietlania danych

Wyswielt wszystkie bazy danych
show databases;
Uzyj baze danych

select database();

Utworz baze danych

create database szkolaDB;

Usun baze danych

DROP DATABASE szkoladb;

Uzyj wybranej bazy danych

use jakas_baza;

Wyswietl dane o wybranej tablicy

describe moja_tablica;

Modyfikacja danych

Stworz tabele

create table dane
(
imie varchar(15) not null,
nazwisko varchar(20) not null,
email varchar(50) null,
ulica varchar(50) null,
miasto varchar(50) null,
data_ur date null,
aktywny int default 1
);

Uzytkownik

Dodaj uzytkownika

CREATE USER 'damian'@'localhost' IDENTIFIED BY 'nowe_haslo';

Zmiana hasla dla uzytkownika

UPDATE mysql.user SET Password=PASSWORD('nowe_haslo') WHERE User='damian';

Usun uzytownika

DROP USER 'damian'@'localhost';

Daj prawa uzytkownika dla bazy danych szkoladb

GRANT ALL PRIVILEGES ON szkoladb.* TO 'damian'@'localhost' WITH GRANT OPTION;
